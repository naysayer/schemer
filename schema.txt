    LANGUAGE plpgsql
    AS $$
             DECLARE
             average_month DECIMAL := 30.4375;
             difference_in_days INTEGER := 0;
             number_of_months DECIMAL;
             monthly_average_result DECIMAL := 0;
             BEGIN
             difference_in_days = (EXTRACT(epoch FROM (SELECT (end_date - start_date)))/86400)::int;
             IF difference_in_days > 0 AND data_to_be_averaged > 0 THEN
             number_of_months = difference_in_days / average_month;
             monthly_average_result = data_to_be_averaged / number_of_months;
             END IF;
             RETURN monthly_average_result;
             END;
             $$;
